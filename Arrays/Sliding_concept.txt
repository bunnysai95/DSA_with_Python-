fixed sliding window
def fixed_sliding_window(arr, k):
    n = len(arr)

    # Edge case
    if n < k:
        return None  # or appropriate value

    # Step 1: Initialize window sum
    window_sum = 0

    # Step 2: Build initial window [0 → k-1]
    for i in range(k):
        window_sum += arr[i]

    # Step 3: Initialize result
    result = window_sum  # max / min / count / etc.

    # Step 4: Slide window
    for i in range(k, n):
        # Add next element, remove previous
        window_sum += arr[i] - arr[i - k]

        # Step 5: Update result
        result = max(result, window_sum)   # change logic as needed

    return result
